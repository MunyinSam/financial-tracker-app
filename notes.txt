# Start only postgres
docker-compose up -d postgres

# Stop only postgres
docker-compose stop postgres

# Start all services
docker-compose up -d

# Stop all services
docker-compose down

# View logs for postgres
docker-compose logs -f postgres

# Check running containers
docker ps

# Restart postgres
docker-compose restart postgres
